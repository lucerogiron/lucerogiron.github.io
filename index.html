class Pickle
def initialize
@cohort = []
end def add_classmate(random)
random.each do |person|
@cohort << person
end
@cohort.each do |classmate|
puts "Welcome #{classmate} to Actualize"
end
return "Let's get started!"
end
def learning()
puts "Check-in question"
@cohort.each do |classmate|
puts "#{classmate} learns more about everyone"
end
puts "Amanda: Let's add testing"
@cohort.each do |classmate|
puts "#{classmate} gets an error :("
end
return "We love pickles and coding"
end
def congrats()
@cohort.each do |classmate|
puts "#{classmate} is a developer!"
end
return "Thanks Amanda, Steve and Beth!"
end
end
random = ["Leah", "Neil", "Josh", "Lucero", "Mike", "Edward", "Luke", "Anna"]
actualize_opp = Pickle.new
p actualize_opp.add_classmate(random)
p actualize_opp.learning p actualize_opp.congrats
